  <script>

    let criminalRecords = JSON.parse(localStorage.getItem('criminalRecords')) || [];


    function saveCriminalRecords() {

      localStorage.setItem('criminalRecords', JSON.stringify(criminalRecords));

    }


    function hideSections() {

      document.querySelectorAll('#viewRecords, #add-section, #edit-section').forEach(div => div.style.display = 'none');

    }


    function requestPassword(correctPassword, sectionId) {

      let userPassword = prompt("Enter Password:");

      if (userPassword === correctPassword) {

        showSection(sectionId);

      } else {

        alert("Incorrect password!");

      }

    }


    function showSection(sectionId) {

      hideSections();

      document.getElementById(sectionId).style.display = 'block';

      if (sectionId === 'viewRecords') displayCriminalRecords();

    }


    function displayCriminalRecords() {

      let criminalList = document.getElementById('criminalList');

      criminalList.innerHTML = '';

      criminalRecords.forEach((record, index) => {

        let li = document.createElement('li');

        li.textContent = `${record.agent} - ${record.name} - ${record.crime} - ${record.group} - ${record.date}`;

        li.onclick = () => loadEditForm(index);

        criminalList.appendChild(li);

      });

    }


    function addCriminalRecord() {

      let agent = document.getElementById('agentName').value.trim();

      let name = document.getElementById('criminalName').value.trim();

      let crime = document.getElementById('crimeType').value;

      let group = document.getElementById('criminalGroup').value;

      let date = document.getElementById('crimeDate').value;

      let report = document.getElementById('crimeReport').value.trim();


      if (agent && name && crime && group && date && report) {

        criminalRecords.push({ agent, name, crime, group, date, report });

        saveCriminalRecords();

        hideSections();

      } else {

        alert('Please fill in all fields.');

      }

    }
